#### masking password

	user@ubuntu:~/lee$ vi lee-2-b.conf
	user@ubuntu:~/lee$ cat lee-2-b.conf
	<source>
	@type forward
	</source>

	<filter lee**>
	@type record_transformer
	enable_ruby
	<record>
		status filtered
		filter processed - ${record["lfs242"]}
		password ${record["password"].gsub(/./,"*")}
	</record>
	remove_keys lfs242
	</filter>

	<match>
	@type stdout
	</match>
